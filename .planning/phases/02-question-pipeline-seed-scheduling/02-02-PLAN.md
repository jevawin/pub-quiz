    Create `.github/workflows/seed-pipeline.yml` with the following exact structure (per D-01, D-02, D-03, D-06, D-08):

    ```yaml
    name: Seed Pipeline

    on:
      schedule:
        - cron: '*/30 * * * *'    # Every 30 minutes for seed phase (~48h)
      workflow_dispatch:
        inputs:
          budget_usd:
            description: 'Budget cap in USD for this seed run'
            required: false
            default: '2.00'
          category_batch_size:
            description: 'Number of new categories to propose'
            required: false
            default: '10'
          knowledge_batch_size:
            description: 'Number of categories to fetch Wikipedia sources for'
            required: false
            default: '20'
          questions_batch_size:
            description: 'Number of questions to generate'
            required: false
            default: '40'

    jobs:
      seed-run:
        runs-on: ubuntu-latest
        timeout-minutes: 30

        steps:
          - name: Checkout repository
            uses: actions/checkout@v4

          - name: Setup Node.js
            uses: actions/setup-node@v4
            with:
              node-version: '22'
              cache: 'npm'
              cache-dependency-path: pipeline/package-lock.json

          - name: Install dependencies
            working-directory: pipeline
            run: npm ci

          - name: Health check -- verify Supabase is reachable
            env:
              SUPABASE_URL: ${{ secrets.SUPABASE_URL }}
              SUPABASE_SERVICE_ROLE_KEY: ${{ secrets.SUPABASE_SERVICE_ROLE_KEY }}
            run: |
              STATUS=$(curl -s -o /dev/null -w "%{http_code}" \
                -H "apikey: $SUPABASE_SERVICE_ROLE_KEY" \
                -H "Authorization: Bearer $SUPABASE_SERVICE_ROLE_KEY" \
                "$SUPABASE_URL/rest/v1/categories?limit=1")
              if [ "$STATUS" = "200" ]; then
                echo "Supabase reachable (HTTP $STATUS)"
              else
                echo "::error::Supabase unreachable (HTTP $STATUS). Project may be paused -- check https://supabase.com/dashboard"
                exit 1
              fi

          - name: Check seed completion threshold
            id: threshold
            working-directory: pipeline
            env:
              SUPABASE_URL: ${{ secrets.SUPABASE_URL }}
              SUPABASE_SERVICE_ROLE_KEY: ${{ secrets.SUPABASE_SERVICE_ROLE_KEY }}
            run: npx tsx src/seed-threshold-check.ts

          - name: Run pipeline (seed batch)
            if: steps.threshold.outputs.seed_complete != 'true'
            working-directory: pipeline
            env:
              ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
              SUPABASE_URL: ${{ secrets.SUPABASE_URL }}
              SUPABASE_SERVICE_ROLE_KEY: ${{ secrets.SUPABASE_SERVICE_ROLE_KEY }}
              PIPELINE_BUDGET_USD: ${{ inputs.budget_usd || '2.00' }}
              CATEGORY_BATCH_SIZE: ${{ inputs.category_batch_size || '10' }}
              KNOWLEDGE_BATCH_SIZE: ${{ inputs.knowledge_batch_size || '20' }}
              QUESTIONS_BATCH_SIZE: ${{ inputs.questions_batch_size || '40' }}
            run: npx tsx src/run-pipeline.ts
    ```

    Key points:
    - Per D-01: Separate workflow file, own cron, own batch sizes, own budget cap
    - Per D-02: `*/30 * * * *` cron (every 30 minutes)
    - Per D-03: Threshold check step with `id: threshold`; pipeline step has `if: steps.threshold.outputs.seed_complete != 'true'`
    - Per D-06: Default batch sizes are 10/20/40 (not 5/10/20 like daily)
    - Per D-08: Default budget is $2.00 (not $1.00 like daily)
    - Per D-04: The existing `question-pipeline.yml` is NOT modified -- verify by diffing after creation
    - The workflow_dispatch inputs mirror the daily workflow but with seed-appropriate defaults
    - The health check step is copied exactly from the daily workflow
    - The `working-directory: pipeline` is set on steps that run in the pipeline directory
