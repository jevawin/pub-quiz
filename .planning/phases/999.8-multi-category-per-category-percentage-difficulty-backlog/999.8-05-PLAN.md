    ```sql
    -- Replace old RPC from 00006 + 00015
    DROP FUNCTION IF EXISTS random_published_questions(TEXT, TEXT, INT);
    DROP FUNCTION IF EXISTS random_published_questions_excluding(TEXT, TEXT, INT, UUID[]);

    CREATE OR REPLACE FUNCTION random_published_questions(
      p_category_slug TEXT,
      p_score_min NUMERIC DEFAULT 0,
      p_score_max NUMERIC DEFAULT 100,
      p_limit INT DEFAULT 10,
      p_exclude_ids UUID[] DEFAULT ARRAY[]::UUID[]
    )
    RETURNS TABLE (
      id UUID,
      question_text TEXT,
      correct_answer TEXT,
      distractors JSONB,
      explanation TEXT,
      category_id UUID,
      category_slug TEXT
    )
    LANGUAGE sql STABLE AS $$
      WITH effective AS (
        SELECT qc.question_id, qc.category_id,
               CASE WHEN qc.observed_n >= 30 THEN qc.observed_score ELSE qc.estimate_score END AS score
        FROM question_categories qc
      ),
      cat_tree AS (
        SELECT id FROM categories WHERE slug = p_category_slug
        UNION ALL
        SELECT c.id FROM categories c JOIN cat_tree t ON c.parent_id = t.id
      )
      SELECT q.id, q.question_text, q.correct_answer, q.distractors, q.explanation,
             e.category_id, c.slug AS category_slug
      FROM questions q
      JOIN effective e ON e.question_id = q.id
      JOIN categories c ON c.id = e.category_id
      WHERE q.status = 'published'
        AND e.score BETWEEN p_score_min AND p_score_max
        AND (p_category_slug IS NULL
             OR p_category_slug = 'general'
             OR e.category_id IN (SELECT id FROM cat_tree))
        AND q.id <> ALL(p_exclude_ids)
      ORDER BY random()
      LIMIT p_limit;
    $$;

    -- Replace 00016 general knowledge RPC
    DROP FUNCTION IF EXISTS random_general_knowledge_questions(TEXT, INT);
    CREATE OR REPLACE FUNCTION random_general_knowledge_questions(
      p_score_min NUMERIC DEFAULT 0,
      p_score_max NUMERIC DEFAULT 100,
      p_limit INT DEFAULT 10,
      p_exclude_ids UUID[] DEFAULT ARRAY[]::UUID[]
    )
    RETURNS TABLE (
      id UUID,
      question_text TEXT,
      correct_answer TEXT,
      distractors JSONB,
      explanation TEXT,
      category_id UUID,
      category_slug TEXT
    )
    LANGUAGE sql STABLE AS $$
      WITH effective AS (
        SELECT qc.question_id, qc.category_id,
               CASE WHEN qc.observed_n >= 30 THEN qc.observed_score ELSE qc.estimate_score END AS score
        FROM question_categories qc
        JOIN categories c ON c.id = qc.category_id
        WHERE c.slug = 'general-knowledge'
      )
      SELECT q.id, q.question_text, q.correct_answer, q.distractors, q.explanation,
             e.category_id, 'general-knowledge'::TEXT AS category_slug
      FROM questions q
      JOIN effective e ON e.question_id = q.id
      WHERE q.status = 'published'
        AND e.score BETWEEN p_score_min AND p_score_max
        AND q.id <> ALL(p_exclude_ids)
      ORDER BY random()
      LIMIT p_limit;
    $$;

    -- Replace 00017 counts RPC
    DROP FUNCTION IF EXISTS counts_by_category_difficulty();
    CREATE OR REPLACE FUNCTION counts_by_category_difficulty()
    RETURNS TABLE (
      category_slug TEXT,
      band TEXT,
      count INT
    )
    LANGUAGE sql STABLE AS $$
      WITH effective AS (
        SELECT qc.question_id, qc.category_id,
               CASE WHEN qc.observed_n >= 30 THEN qc.observed_score ELSE qc.estimate_score END AS score
        FROM question_categories qc
      ),
      banded AS (
        SELECT
          e.category_id,
          CASE
            WHEN e.score <= 33 THEN 'hard'
            WHEN e.score <= 66 THEN 'medium'
            ELSE 'easy'
          END AS band,
          e.question_id
        FROM effective e
        JOIN questions q ON q.id = e.question_id AND q.status = 'published'
      )
      SELECT c.slug AS category_slug, b.band, COUNT(DISTINCT b.question_id)::INT AS count
      FROM banded b
      JOIN categories c ON c.id = b.category_id
      GROUP BY c.slug, b.band
      ORDER BY c.slug, b.band;
    $$;

    GRANT EXECUTE ON FUNCTION random_published_questions(TEXT, NUMERIC, NUMERIC, INT, UUID[]) TO anon, authenticated;
    GRANT EXECUTE ON FUNCTION random_general_knowledge_questions(NUMERIC, NUMERIC, INT, UUID[]) TO anon, authenticated;
    GRANT EXECUTE ON FUNCTION counts_by_category_difficulty() TO anon, authenticated;
    ```

    ```typescript
    import { supabase } from './supabase';

    export interface RpcQuestionRow {
      id: string;
      question_text: string;
      correct_answer: string;
      distractors: string[];
      explanation: string;
      category_id: string;
      category_slug: string;
    }

    export interface FetchRandomQuestionsArgs {
      categorySlug: string | null;  // null or 'general' for general knowledge
      scoreMin: number;
      scoreMax: number;
      limit: number;
      excludeIds?: string[];
    }

    export async function fetchRandomQuestions(args: FetchRandomQuestionsArgs): Promise<RpcQuestionRow[]> {
      const isGk = args.categorySlug === null || args.categorySlug === 'general-knowledge' || args.categorySlug === 'general';
      const rpcName = isGk ? 'random_general_knowledge_questions' : 'random_published_questions';
      const params: Record<string, unknown> = {
        p_score_min: args.scoreMin,
        p_score_max: args.scoreMax,
        p_limit: args.limit,
        p_exclude_ids: args.excludeIds ?? [],
      };
      if (!isGk) params.p_category_slug = args.categorySlug;

      const { data, error } = await supabase.rpc(rpcName, params);
      if (error) throw error;
      return (data ?? []) as RpcQuestionRow[];
    }
    ```
