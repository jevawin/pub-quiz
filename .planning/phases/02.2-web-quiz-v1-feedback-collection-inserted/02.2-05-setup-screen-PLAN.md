Create `apps/web/src/lib/questions.ts`:
```ts
import { supabase } from './supabase';
import { shuffle } from './shuffle';
import { uiToScoreRange, type UiDifficulty } from './difficulty';
import type { LoadedQuestion } from '@/state/quiz';

type RpcRow = {
  id: string;
  question_text: string;
  correct_answer: string;
  distractors: string[];
  explanation: string | null;
  category_id: string;
  category_slug: string;
};

export async function fetchRandomQuestions(
  uiDifficulty: UiDifficulty,
  categorySlugs: string[],
  n: number,
  excludeIds: string[] = [],
): Promise<LoadedQuestion[]> {
  const { min, max } = uiToScoreRange(uiDifficulty);
  const isGeneral = categorySlugs.length === 0 || categorySlugs.includes('general');
  const rpcName = isGeneral ? 'random_general_knowledge_questions' : 'random_published_questions_excluding';
  const params: Record<string, unknown> = isGeneral
    ? { p_score_min: min, p_score_max: max, p_limit: n }
    : {
        p_score_min: min,
        p_score_max: max,
        p_category_slug: categorySlugs[0],
        p_limit: n,
        p_exclude_ids: excludeIds,
      };
  const { data, error } = await supabase.rpc(rpcName, params);
  if (error) throw error;
  const rows = (data ?? []) as RpcRow[];
  return rows.map((r) => {
    const distractors = Array.isArray(r.distractors) ? r.distractors : [];
    const options = shuffle([r.correct_answer, ...distractors]);
    const correctIndex = options.indexOf(r.correct_answer);
    return {
      id: r.id,
      question_text: r.question_text,
      options,
      correctIndex,
      explanation: r.explanation,
    };
  });
}
```
